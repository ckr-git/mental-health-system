```sql

-- 删除测试账号
DELETE FROM user WHERE username IN ('patient001', 'patient002', 'patient003', 'doctor001', 'doctor002', 'admin');
DELETE FROM doctor_info WHERE user_id NOT IN (SELECT id FROM user);
DELETE FROM symptom_record WHERE user_id NOT IN (SELECT id FROM user);
```
